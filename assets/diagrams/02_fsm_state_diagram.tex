% ROUND 2: 3-state FSM (INIT/MSG/SLEEP), category-based navigation.
% Each key is a FIXED jump to its category head; TIMEOUT advances
% sequentially within the category, falling back to Default at a
% non-emergency category's end, sticking forever in Emergency. The 4 key
% targets are listed in a side legend rather than as 4 separate crowded
% self-loops on one small node.
\begin{tikzpicture}[
  every node/.style={font=\sffamily},
  state/.style={circle, minimum size=2.0cm, line width=1.6pt,
                align=center, font=\normalsize\bfseries},
  initstate/.style={state, draw=InitBorder, fill=InitFill,
                    text=InitBorder},
  msgstate/.style={state, draw=MsgBorder, fill=MsgFill,
                   text=MsgBorder},
  sleepstate/.style={state, draw=SleepBorder, fill=SleepFill,
                     text=SleepBorder},
  autoarr/.style={-{Latex[length=2.7mm,width=1.9mm]}, draw=ArrowGray,
                  line width=1.25pt},
  btnarr/.style={-{Latex[length=2.7mm,width=1.9mm]}, draw=BtnBlue,
                 line width=1.35pt},
  timeoutarr/.style={-{Latex[length=2.8mm,width=2.0mm]}, draw=TimeoutRed,
                     line width=1.5pt},
  lbl/.style={font=\scriptsize\bfseries, fill=white, inner sep=0.8pt},
  smalllbl/.style={font=\tiny\bfseries, inner sep=0.4pt}
]

\node[initstate] (INIT)  at (0.0, 0.0)  {INIT};
\node[msgstate]  (MSG)   at (5.2, 0.0)  {MSG};
\node[sleepstate](SLEEP) at (11.0, 0.0) {SLEEP};

% reset is an automatic startup transition, not a timer expiry, so it is drawn
% gray to match the legend (gray = automatic startup), like 'auto'.
\draw[autoarr] ([xshift=-1.35cm]INIT.west) --
  node[lbl, above, text=ArrowGray] {reset} (INIT.west);
\draw[autoarr] (INIT.east) --
  node[lbl, above, text=ArrowGray] {auto} (MSG.west);

% MSG <-> SLEEP: two parallel offset arrows.
\draw[timeoutarr] ([yshift=0.30cm]MSG.east) --
  node[smalllbl, above=1pt, text=TimeoutRed, align=center]
       {sleep\_timeout\\(DEFAULT only)}
  ([yshift=0.30cm]SLEEP.west);
\draw[btnarr] ([yshift=-0.30cm]SLEEP.west) --
  node[smalllbl, below=1pt, text=BtnBlue, align=center]
       {any key $\to$\\category head}
  ([yshift=-0.30cm]MSG.east);

% MSG self-loop (top): any of KEY0-3 -- fixed jump, see legend below.
\draw[btnarr] (MSG.north west)
  to[out=120, in=60, looseness=8]
  node[lbl, above=2pt, text=BtnBlue, align=center]
       {KEY0-3: fixed jump to\\that key's category head (see legend)}
  (MSG.north east);

% MSG self-loop (bottom): TIMEOUT advances within the current category.
\draw[timeoutarr] (MSG.south west)
  to[out=240, in=300, looseness=8]
  node[lbl, below=2pt, text=TimeoutRed, align=center]
       {timeout: next in category\\(non-emergency last $\to$ DEFAULT;\\EMERGENCY sticks)}
  (MSG.south east);

% Legend: which key maps to which category head.
\node[draw=GrayBorder, rounded corners=4pt, fill=GrayBg, align=left,
      font=\small, inner sep=8pt, anchor=north]
  at (5.2, -5.3) {%
    \textbf{Key} \hspace{0.4cm} \textbf{Category head}\\[2pt]
    \textcolor{BtnBlue}{KEY0} \hspace{0.9cm} EXERCISE (msg 3)\\
    \textcolor{BtnBlue}{KEY1} \hspace{0.9cm} SESSION (msg 8)\\
    \textcolor{BtnBlue}{KEY2} \hspace{0.9cm} EMERGENCY (msg 16)\\
    \textcolor{BtnBlue}{KEY3} \hspace{0.9cm} DEFAULT (msg 0) -- also escapes a stuck EMERGENCY
  };

\end{tikzpicture}
